⚠️ Common Vulnerabilities

Critical security issues in smart contract development with real-world examples and mitigations.

01

Reentrancy Attack

External calls allow recursive exploitation

Critical

An attacker exploits external calls that callback into the vulnerable contract before state updates complete. This enables draining funds through recursive calls.

Real-world Impact: The DAO hack (2016) — $60M stolen using this exact vulnerability.

Types of Reentrancy:

  • Single-function: Attacking the same function recursively
  • Cross-function: Attacking different functions that share state
  • Cross-contract: Exploiting reentrancy across multiple contracts
  • Read-only: Exploiting view functions during reentrancy state
❌ VULNERABLE 
contract VulnerableBank {
    mapping(address => uint) public balances;

    function withdraw(uint amount) external {
        require(balances[msg.sender] >= amount, "Insufficient");

        // External call BEFORE state update - DANGEROUS!
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");

        // Attacker's receive() calls withdraw() again
        // before this line executes
        balances[msg.sender] -= amount;
    }
}

// Attacker Contract
contract Attacker {
    VulnerableBank public bank;

    function attack() external payable {
        bank.deposit{value: msg.value}();
        bank.withdraw(msg.value);
    }

    receive() external payable {
        if (address(bank).balance >= 1 ether) {
            bank.withdraw(1 ether); // Re-enter!
        }
    }
}
✅ SECURE - Checks-Effects-Interactions 
contract SecureBank {
    mapping(address => uint) public balances;

    function withdraw(uint amount) external {
        // 1. CHECKS
        require(balances[msg.sender] >= amount, "Insufficient");

        // 2. EFFECTS - Update state BEFORE external call
        balances[msg.sender] -= amount;

        // 3. INTERACTIONS - External call LAST
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");
    }
}
✅ SECURE - Using ReentrancyGuard 
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

contract SecureBankWithGuard is ReentrancyGuard {
    mapping(address => uint) public balances;

    function withdraw(uint amount) external nonReentrant {
        require(balances[msg.sender] >= amount);
        balances[msg.sender] -= amount;
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success);
    }

    function transfer(address to, uint amount) external nonReentrant {
        require(balances[msg.sender] >= amount);
        balances[msg.sender] -= amount;
        balances[to] += amount;
    }
}

🛡️ Prevention

  • Always use Checks-Effects-Interactions pattern
  • Use ReentrancyGuard for all state-changing functions
  • Consider nonReentrant on view functions in DeFi protocols
  • Be aware of cross-contract reentrancy in multi-contract systems
02

Integer Overflow/Underflow

Arithmetic operations wrap around silently

Critical

In Solidity < 0.8.0, arithmetic operations that exceed type bounds wrap around to unexpected values, enabling attackers to mint unlimited tokens or bypass balance checks.

Real-world Impact: BEC Token (2018) — Unlimited token minting via overflow.
❌ VULNERABLE (Solidity < 0.8.0) 
contract VulnerableToken {
    mapping(address => uint256) public balances;

    // Overflow: receivers.length * value can wrap to small number
    function batchTransfer(address[] memory receivers, uint256 value) public {
        uint256 total = receivers.length * value; // Can overflow!
        require(balances[msg.sender] >= total);

        balances[msg.sender] -= total;
        for (uint i = 0; i < receivers.length; i++) {
            balances[receivers[i]] += value;
        }
    }

    // Underflow: 0 - 1 = MAX_UINT256
    function withdraw(uint256 amount) public {
        balances[msg.sender] -= amount; // If balance is 0, becomes huge!
        payable(msg.sender).transfer(amount);
    }
}

// Example: uint8 max = 255
// 255 + 1 = 0 (overflow)
// 0 - 1 = 255 (underflow)
✅ SECURE - Solidity >= 0.8.0 
// Solidity >= 0.8.0 has automatic overflow checks
contract SecureToken {
    mapping(address => uint256) public balances;

    function batchTransfer(address[] memory receivers, uint256 value) public {
        uint256 total = receivers.length * value; // Reverts on overflow
        require(balances[msg.sender] >= total);

        balances[msg.sender] -= total; // Reverts on underflow
        for (uint i = 0; i < receivers.length; i++) {
            balances[receivers[i]] += value;
        }
    }
}

// For legacy code, use SafeMath:
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract LegacySecure {
    using SafeMath for uint256;

    function transfer(address to, uint256 amount) public {
        balances[msg.sender] = balances[msg.sender].sub(amount);
        balances[to] = balances[to].add(amount);
    }
}
⚠️ CAREFUL - Unchecked blocks 
// Only use unchecked when mathematically proven safe
contract OptimizedLoop {
    function sum(uint256[] calldata arr) external pure returns (uint256 total) {
        uint256 length = arr.length;
        for (uint256 i = 0; i < length;) {
            total += arr[i];
            unchecked { ++i; } // Safe: can't overflow before gas limit
        }
    }

    // ❌ DANGEROUS
    function unsafeDecrement(uint256 x) external pure returns (uint256) {
        unchecked { return x - 1; } // Can underflow if x == 0!
    }
}

🛡️ Prevention

  • Use Solidity >= 0.8.0 for automatic overflow checks
  • Use SafeMath library for older versions
  • Only use unchecked when mathematically proven safe
  • Be extra careful with multiplication before division
03

Access Control Vulnerabilities

Missing or incorrect permission checks

Critical

Missing or incorrect access controls allow unauthorized users to execute privileged functions, potentially taking over the entire contract.

Real-world Impact: Parity Multisig Wallet (2017) — $30M stolen, $150M frozen.
❌ VULNERABLE 
contract VulnerableWallet {
    address public owner;

    constructor() { owner = msg.sender; }

    // Anyone can call this!
    function setOwner(address _newOwner) public {
        owner = _newOwner;
    }

    // Anyone can withdraw!
    function withdraw() public {
        payable(msg.sender).transfer(address(this).balance);
    }
}

// Unprotected initializer (proxy pattern)
contract VulnerableImplementation {
    address public owner;
    bool public initialized;

    // Anyone can call initialize on the implementation!
    function initialize(address _owner) external {
        require(!initialized);
        owner = _owner;
        initialized = true;
    }
}
✅ SECURE - Role-based Access 
import "@openzeppelin/contracts/access/AccessControl.sol";

contract SecureProtocol is AccessControl {
    bytes32 public constant ADMIN_ROLE = keccak256("ADMIN_ROLE");
    bytes32 public constant OPERATOR_ROLE = keccak256("OPERATOR_ROLE");

    constructor() {
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(ADMIN_ROLE, msg.sender);
    }

    function adminFunction() external onlyRole(ADMIN_ROLE) {
        // Only admins can call
    }

    function operatorFunction() external onlyRole(OPERATOR_ROLE) {
        // Only operators can call
    }
}

// Protected initializer for proxies
import "@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol";

contract SecureImplementation is Initializable {
    address public owner;

    /// @custom:oz-upgrades-unsafe-allow constructor
    constructor() { _disableInitializers(); }

    function initialize(address _owner) external initializer {
        owner = _owner;
    }
}

🛡️ Prevention

  • Always use access control modifiers on sensitive functions
  • Use OpenZeppelin's Ownable or AccessControl
  • Protect initializers in upgradeable contracts
  • Use _disableInitializers() in implementation constructors
  • Review function visibility carefully (public vs external vs internal)
04

tx.origin Authentication

Phishing attacks via contract intermediary

High

Using tx.origin for authentication allows phishing attacks where a malicious contract tricks users into executing transactions that drain their wallets.

❌ VULNERABLE 
contract VulnerableWallet {
    address public owner;

    function transfer(address to, uint amount) public {
        // tx.origin is the EOA that started the transaction
        require(tx.origin == owner, "Not owner");
        payable(to).transfer(amount);
    }
}

// Attacker contract
contract PhishingAttack {
    VulnerableWallet wallet;
    address attacker;

    constructor(address _wallet) {
        wallet = VulnerableWallet(_wallet);
        attacker = msg.sender;
    }

    // Victim calls this thinking it's legitimate
    function claimReward() external {
        // tx.origin is still the victim!
        wallet.transfer(attacker, address(wallet).balance);
    }
}
✅ SECURE - Use msg.sender 
contract SecureWallet {
    address public owner;

    function transfer(address to, uint amount) public {
        require(msg.sender == owner, "Not owner");
        payable(to).transfer(amount);
    }
}

// Acceptable use: prevent contracts from calling
contract NoContractCalls {
    modifier noContracts() {
        require(tx.origin == msg.sender, "No contracts");
        _;
    }

    function sensitiveFunction() external noContracts {
        // Only EOAs can call
    }
}

🛡️ Prevention

  • Never use tx.origin for authentication
  • Always use msg.sender for access control
  • Only use tx.origin to check if caller is an EOA

Unsafe Delegatecall

External code modifies your storage

Critical

delegatecall executes code in the context of the calling contract, using the caller's storage. This can lead to storage corruption or complete contract takeover.

Real-world Impact: Parity Wallet (2017) — $150M frozen forever.
❌ VULNERABLE 
contract VulnerableProxy {
    address public owner;        // Slot 0
    address public implementation; // Slot 1

    // Anyone can call with arbitrary address!
    function execute(address target, bytes calldata data) external {
        target.delegatecall(data); // Attacker's code with OUR storage
    }
}

// Attacker deploys this and calls execute()
contract MaliciousImpl {
    address public owner; // Maps to slot 0 of proxy

    function attack() external {
        owner = msg.sender; // Overwrites proxy's owner!
    }
}

// Storage collision in proxy pattern
contract ImplementationV1 {
    uint256 public value; // Slot 0 - COLLIDES with proxy's owner!

    function setValue(uint256 _v) external {
        value = _v; // Overwrites proxy's storage!
    }
}
✅ SECURE - OpenZeppelin Proxy 
import "@openzeppelin/contracts/proxy/ERC1967/ERC1967Proxy.sol";
import "@openzeppelin/contracts-upgradeable/proxy/utils/UUPSUpgradeable.sol";

contract SecureImplementation is UUPSUpgradeable {
    // Storage starts AFTER proxy reserved slots
    uint256 public value;

    function _authorizeUpgrade(address newImpl) internal override {
        // Add access control here
    }
}

// Unstructured storage pattern avoids collisions
contract UnstructuredStorageProxy {
    // Random slot location
    bytes32 private constant IMPL_SLOT =
        bytes32(uint256(keccak256("eip1967.proxy.implementation")) - 1);

    function _getImplementation() internal view returns (address impl) {
        bytes32 slot = IMPL_SLOT;
        assembly { impl := sload(slot) }
    }
}

🛡️ Prevention

  • Never delegatecall to user-controlled addresses
  • Use established proxy patterns (ERC-1967, UUPS, Transparent)
  • Be extremely careful with storage layout in upgradeable contracts
  • Use OpenZeppelin's upgradeable contracts library

Front-Running & MEV

Transaction ordering exploitation

High

Attackers observe pending transactions in the mempool and submit their own with higher gas to execute first, extracting value through sandwich attacks and displacement.

❌ VULNERABLE 
contract VulnerableDEX {
    function swap(address tokenIn, address tokenOut, uint amountIn) external {
        // Attacker sees this in mempool
        // Front-runs with their own swap
        // Back-runs after victim's swap at worse price
        uint price = getPrice(tokenIn, tokenOut);
        // Execute swap...
    }
}

contract VulnerableAuction {
    function submitSolution(bytes32 answer) external {
        // Attacker sees the answer in mempool
        // Submits with higher gas to win
        require(keccak256(abi.encode(answer)) == puzzle);
        winner = msg.sender;
    }
}
✅ MITIGATED - Commit-Reveal 
contract SecureAuction {
    mapping(address => bytes32) public commits;
    uint256 public commitDeadline;
    uint256 public revealDeadline;

    // Phase 1: Commit (hidden)
    function commit(bytes32 hashedBid) external {
        require(block.timestamp < commitDeadline);
        commits[msg.sender] = hashedBid;
    }

    // Phase 2: Reveal (after commit phase)
    function reveal(uint256 bidAmount, bytes32 salt) external {
        require(block.timestamp >= commitDeadline);
        require(block.timestamp < revealDeadline);

        bytes32 hash = keccak256(abi.encodePacked(
            bidAmount, salt, msg.sender
        ));
        require(commits[msg.sender] == hash, "Invalid");
        // Process bid...
    }
}

// Slippage protection for DEX
contract ProtectedDEX {
    function swap(
        address tokenIn,
        address tokenOut,
        uint amountIn,
        uint minAmountOut,  // Slippage protection
        uint deadline       // Time protection
    ) external {
        require(block.timestamp <= deadline, "Expired");
        uint amountOut = calculateOutput(tokenIn, tokenOut, amountIn);
        require(amountOut >= minAmountOut, "Slippage");
        // Execute...
    }
}

🛡️ Prevention

  • Use commit-reveal patterns for sensitive operations
  • Implement slippage protection and deadlines
  • Use batch auctions for fair ordering
  • Consider private mempools (Flashbots Protect)

Denial of Service (DoS)

Blocking legitimate users

High

Attackers prevent legitimate users from using the contract by causing functions to fail or consume excessive gas through unbounded loops or external call failures.

❌ VULNERABLE 
contract VulnerableAirdrop {
    address[] public recipients;

    function airdrop(uint amount) external {
        for (uint i = 0; i < recipients.length; i++) {
            // If ANY transfer fails, entire airdrop fails
            // Attacker adds contract that reverts in receive()
            payable(recipients[i]).transfer(amount);
        }
    }
}

contract VulnerableVoting {
    address[] public voters;

    function countVotes() external view returns (uint) {
        uint count = 0;
        // Gas limit exceeded if too many voters
        for (uint i = 0; i < voters.length; i++) {
            if (hasVoted[voters[i]]) count++;
        }
        return count;
    }
}
✅ SECURE - Pull Pattern 
contract SecureAirdrop {
    mapping(address => uint) public claimable;

    function setClaimable(address[] calldata recipients, uint amount) external {
        for (uint i = 0; i < recipients.length; i++) {
            claimable[recipients[i]] += amount;
        }
    }

    // Users pull their own funds
    function claim() external {
        uint amount = claimable[msg.sender];
        require(amount > 0, "Nothing to claim");
        claimable[msg.sender] = 0;
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success);
    }
}

// Paginated processing
contract SecureVoting {
    uint public lastProcessedIndex;

    function countVotesBatch(uint batchSize) external returns (bool done) {
        uint end = lastProcessedIndex + batchSize;
        if (end > voterCount) end = voterCount;

        for (uint i = lastProcessedIndex; i < end; i++) {
            // Process batch...
        }
        lastProcessedIndex = end;
        return lastProcessedIndex >= voterCount;
    }
}

🛡️ Prevention

  • Use Pull over Push pattern for payments
  • Implement pagination for loops
  • Set maximum batch sizes
  • Avoid unbounded arrays in storage

Oracle & Price Manipulation

Flash loan price attacks

Critical

Attackers manipulate on-chain price data using flash loans to temporarily skew prices, enabling exploitation of DeFi lending and trading protocols.

Real-world Impact: Numerous DeFi hacks totaling hundreds of millions in losses.
❌ VULNERABLE - Spot Price 
contract VulnerableLending {
    IUniswapV2Pair public pair;

    function getPrice() public view returns (uint) {
        (uint reserve0, uint reserve1, ) = pair.getReserves();
        // Spot price can be manipulated with flash loans!
        return reserve0 * 1e18 / reserve1;
    }

    function borrow(uint collateral) external {
        uint price = getPrice(); // Manipulated!
        uint borrowLimit = collateral * price / 1e18;
        // Attacker deposits, manipulates price up,
        // borrows max, price returns, profit!
    }
}
✅ SECURE - Chainlink Oracle 
import "@chainlink/contracts/src/v0.8/interfaces/AggregatorV3Interface.sol";

contract SecureLending {
    AggregatorV3Interface internal priceFeed;
    uint256 public constant STALENESS = 1 hours;

    function getLatestPrice() public view returns (int) {
        (
            uint80 roundId,
            int price,
            uint startedAt,
            uint updatedAt,
            uint80 answeredInRound
        ) = priceFeed.latestRoundData();

        // Validate the data
        require(price > 0, "Invalid price");
        require(updatedAt > 0, "Round not complete");
        require(block.timestamp - updatedAt < STALENESS, "Stale");
        require(answeredInRound >= roundId, "Stale round");

        return price;
    }
}

🛡️ Prevention

  • Never use spot prices for critical operations
  • Implement TWAP (Time-Weighted Average Price)
  • Use established oracles (Chainlink, Band Protocol)
  • Validate oracle data freshness and bounds
  • Implement circuit breakers for extreme price movements

Signature Replay Attacks

Reusing valid signatures maliciously

High

Valid signatures can be reused maliciously on the same contract, different chains, or different contracts if proper replay protection is not implemented.

❌ VULNERABLE 
contract VulnerablePermit {
    function permitTransfer(
        address from,
        address to,
        uint amount,
        bytes memory signature
    ) external {
        bytes32 hash = keccak256(abi.encodePacked(from, to, amount));
        address signer = recoverSigner(hash, signature);
        require(signer == from, "Invalid");

        // Same signature can be used multiple times!
        // And on other chains where contract is deployed!
        balances[from] -= amount;
        balances[to] += amount;
    }
}
✅ SECURE - EIP-712 with Nonce 
contract SecurePermit {
    mapping(address => uint) public nonces;
    bytes32 public immutable DOMAIN_SEPARATOR;

    bytes32 public constant PERMIT_TYPEHASH = keccak256(
        "Permit(address from,address to,uint256 amount,uint256 nonce,uint256 deadline)"
    );

    constructor() {
        DOMAIN_SEPARATOR = keccak256(abi.encode(
            keccak256("EIP712Domain(string name,string version,uint256 chainId,address verifyingContract)"),
            keccak256(bytes("SecurePermit")),
            keccak256(bytes("1")),
            block.chainid,      // Chain protection
            address(this)       // Contract protection
        ));
    }

    function permitTransfer(
        address from, address to, uint amount,
        uint deadline, uint8 v, bytes32 r, bytes32 s
    ) external {
        require(block.timestamp <= deadline, "Expired");

        bytes32 structHash = keccak256(abi.encode(
            PERMIT_TYPEHASH, from, to, amount,
            nonces[from]++,  // Increment nonce
            deadline
        ));

        bytes32 digest = keccak256(abi.encodePacked(
            "\x19\x01", DOMAIN_SEPARATOR, structHash
        ));

        address signer = ecrecover(digest, v, r, s);
        require(signer != address(0) && signer == from);
        // Transfer...
    }
}

🛡️ Prevention

  • Include nonces in signed messages
  • Include chain ID (EIP-712 domain separator)
  • Include contract address in domain
  • Add deadline/expiration to signatures
  • Use OpenZeppelin's EIP-712 implementation

Unsafe External Calls

Unchecked return values and gas issues

High

Improper handling of external calls can lead to silent failures, unexpected behavior, or reentrancy vulnerabilities.

✅ SECURE - Safe Token Transfer 
import "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";

contract SafeTokenContract {
    using SafeERC20 for IERC20;

    function transferTokens(IERC20 token, address to, uint amount) external {
        // Handles non-standard tokens (USDT, etc.)
        token.safeTransfer(to, amount);
    }
}

// Safe ETH transfer
function sendETH(address to, uint amount) internal {
    (bool success, ) = to.call{value: amount}("");
    require(success, "Transfer failed");
}

// Gas-limited callback
function safeCallback(address target) external {
    (bool success, ) = target.call{gas: 50000}(
        abi.encodeWithSignature("callback()")
    );
    // Continue even if fails
}

Timestamp Dependence

Miner-manipulable block.timestamp

Medium

Miners can manipulate block.timestamp within ~15 seconds, making it unreliable for randomness or exact comparisons.

✅ SECURE - Longer Windows 
contract SaferTimestamp {
    uint public constant MIN_DELAY = 1 days;
    uint public executeTime;

    function schedule() external {
        executeTime = block.timestamp + MIN_DELAY;
    }

    function execute() external {
        // Use >= not ==, longer windows
        require(block.timestamp >= executeTime, "Too early");
    }
}

// Or use block numbers for precision
contract BlockBasedTiming {
    uint public constant BLOCKS_PER_DAY = 7200;
    uint public unlockBlock;

    function lock() external {
        unlockBlock = block.number + BLOCKS_PER_DAY;
    }
}

Weak Randomness

Predictable on-chain sources

High

On-chain randomness sources (block.timestamp, block.difficulty, blockhash) are predictable and manipulable.

✅ SECURE - Chainlink VRF 
import "@chainlink/contracts/src/v0.8/vrf/VRFConsumerBaseV2.sol";

contract SecureRandom is VRFConsumerBaseV2 {
    VRFCoordinatorV2Interface COORDINATOR;
    mapping(uint256 => uint256) public requestToRandom;

    function requestRandom() external returns (uint256) {
        return COORDINATOR.requestRandomWords(
            keyHash, subscriptionId, 3, 100000, 1
        );
    }

    function fulfillRandomWords(
        uint256 requestId,
        uint256[] memory randomWords
    ) internal override {
        requestToRandom[requestId] = randomWords[0];
    }
}

Force Feeding Ether

ETH sent without triggering receive()

Medium

Ether can be forced into contracts via selfdestruct or mining rewards without triggering receive(), breaking balance-dependent logic.

✅ SECURE - Track Deposits 
contract SecureGame {
    uint public totalDeposits; // Track separately

    function play() external payable {
        require(msg.value == 1 ether);
        totalDeposits += msg.value;

        // Use tracked variable, not address(this).balance
        if (totalDeposits == 10 ether) {
            winner = msg.sender;
        }
    }
}

Storage Collision

Upgradeable contract storage corruption

Critical

Changing storage layout in upgradeable contracts corrupts existing data. Variables must only be appended, never removed or reordered.

✅ SECURE - Storage Gaps 
contract ImplementationV1 {
    uint256 public value;    // Slot 0
    address public owner;    // Slot 1
    uint256[48] private __gap; // Reserve slots
}

contract ImplementationV2 {
    uint256 public value;    // Slot 0 - Same
    address public owner;    // Slot 1 - Same
    uint256 public newVar;   // Slot 2 - Uses gap
    uint256[47] private __gap; // Reduced by 1
}

// Never remove, never reorder!
// Only append at the end

🛡️ Security Patterns

Battle-tested patterns to secure your smart contracts.

Checks-Effects-Interactions

Validate inputs, update state, then make external calls. Prevents reentrancy and ensures consistent state.

Pull over Push

Let users withdraw funds instead of pushing payments. Prevents DoS from failing transfers.

Emergency Stop

Circuit breaker pattern with Pausable. Allows halting operations during incidents.

Rate Limiting

Cooldown periods between actions. Limits damage from compromised accounts.

Commit-Reveal

Two-phase submission hides data until reveal. Prevents front-running.

Guard Check

Modifiers for common validations. Clean, reusable input validation.

Combined Pattern Example 
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract SecureVault is ReentrancyGuard, Pausable, Ownable {
    mapping(address => uint) public balances;
    mapping(address => uint) public lastAction;
    uint public constant COOLDOWN = 1 hours;

    error ZeroAmount();
    error Cooldown();

    modifier rateLimit() {
        if (block.timestamp < lastAction[msg.sender] + COOLDOWN)
            revert Cooldown();
        lastAction[msg.sender] = block.timestamp;
        _;
    }

    function withdraw(uint amount)
        external
        nonReentrant
        whenNotPaused
        rateLimit
    {
        // Checks
        if (amount == 0) revert ZeroAmount();
        require(balances[msg.sender] >= amount);

        // Effects
        balances[msg.sender] -= amount;

        // Interactions
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success);
    }

    function pause() external onlyOwner { _pause(); }
    function unpause() external onlyOwner { _unpause(); }
}

✨ Best Practices

Code quality guidelines for secure, maintainable contracts.

Variables & Visibility

Variable Best Practices 
// Always specify visibility
uint256 private _balance;      // Only this contract
uint256 internal _rate;        // This + inherited
uint256 public totalSupply;    // Public (creates getter)

// Immutable & constant save gas
address public immutable FACTORY;   // Set once in constructor
uint256 public constant FEE = 30;   // Compile-time constant

// Pack structs for storage efficiency
struct User {
    uint128 balance;    // Slot 0 (16 bytes)
    uint64 timestamp;   // Slot 0 (8 bytes)
    uint32 nonce;       // Slot 0 (4 bytes)
    bool active;        // Slot 0 (1 byte)
    address addr;       // Slot 1
}

Functions

Function Declaration Order 
function example(uint256 amount)

// Cache storage in memory for loops
function sum() external view returns (uint256 total) {
    uint256 len = items.length; // Cache
    for (uint256 i = 0; i < len;) {
        total += items[i];
        unchecked { ++i; } // Safe increment
    }
}

// Storage packing (32 byte slots)
// Bad: 3 slots
uint256 a;  // Slot 0
uint8 b;    // Slot 1
uint256 c;  // Slot 2

// Good: 2 slots
uint256 a;  // Slot 0
uint256 c;  // Slot 1
uint8 b;    // Slot 1 (packed)
